#include <iostream>
using namespace std;

template <typename T>
void Swap(T& a, T& b);

template <typename T> //新模板
void Swap02(T a[], T b[], int);

int main()
{
	int a = 10, b = 20;
	int c[4] = { 1, 2, 3, 4 }, d[4] = { 5, 6, 7, 8 };
	Swap(a, b);
	Swap02(c, d, 3);
	cout << "a = " << a << " " << "b = " << b << endl;
	cout << "c = ";
	for (int i = 0; i < 4; i++)
	{
		cout << c[i] << " ";
	}
	cout << endl << "d = ";
	for (int i = 0; i < 4; i++)
	{
		cout << d[i] << " ";
	}
	return 0;
}

template <typename T>
void Swap(T& a, T& b)
{
	T temp;
	temp = a;
	a = b;
	b = temp;
}

template <typename T>
void Swap02(T a[], T b[], int n)
{
	T temp;
	for (int i = 0; i < n; i++)
	{
		temp = a[i];
		a[i] = b[i];
		b[i] = temp;
	}
}
