`template <typename/class T>`
自动推导类型
```
#include <iostream>
using namespace std;

template <typename T>

void Swap(T& a, T& b)
{
	T temp;
	temp = a;
	a = b;
	b = temp;
}

int main()
{
	int a = 10, b = 20;
	cout << a << " " << b << endl;
	Swap(a, b);
	cout << a << " " << b << endl;
	double c = 1.3, d = 5.2;
	cout << c << " " << d << endl;
	Swap(c, d);
	cout << c << " " << d << endl;
	return 0;
}
```
1.必须推导出**一致的数据类型T**

2.必须确定T的数据类型

模板用例：排序函数
```
#include <iostream>
using namespace std;

template <typename T>
void Sort(T arr[], int len)
{
	for (int i = 0; i < len; i++)
	{
		int max = i;
		for (int j = i + 1; j < len; j++)
		{
			if (arr[max] < arr[j])
			{
				max = j;
			}
		}
		if (max != i)
		{
			T temp;
			temp = arr[max];
			arr[max] = arr[i];
			arr[i] = temp;
		}
	}
}

int main()
{
	char c[5] = "awaw"; //4字母？
	Sort(c, 5);
	for (int i = 0; i < 5; i++)
	{
		cout << c[i];
	}
	int n[5] = { 1, 4, 6, 2, 2 };
	Sort(n, 5);
	for (int i = 0; i < 5; i++)
	{
		cout << n[i];
	}
	return 0;
}
```
