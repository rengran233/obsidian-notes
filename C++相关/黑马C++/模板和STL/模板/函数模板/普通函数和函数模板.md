普通函数和函数模板的区别
1.普通函数可以发生**隐式类型转换** 
2.函数模板用自动类型推导不行，显式指定类型可以
```
#include <iostream>
using namespace std;

int add01(int a, int b)
{
	return a + b;
}

template <typename T>
int add02 (T a, T b)
{
	return a + b;
}

template <typename T1, typename T2>
int add03(T1& a, T2& b)
{
	return a + b;
}

int main()
{
	int a = 10;
	int b = 20;
	char c = 'c';
	cout << add01(a, b) << endl;
	cout << add01(a, c) << endl;//ASCII码
	cout << add02(a, b) << endl;
	//cout << add02(a, c) << endl;
	cout << add02<int> (a, c) << endl; //指针/引用不可隐式转换
	cout << add03(a, c) << endl;
	return 0;
}
```


普通函数和函数模板的调用规则
1.如果普通函数和函数模，优先调用普通函数
2.可以通过空函数列表强行调用函数模板
3.模板函数可以发生重载
4.如果函数模板更匹配，优先调用函数模板
```
#include <iostream>
using namespace std;

void func(int)
{
	cout << "调用普通函数" << endl;
}

template <typename T>
void func(T)
{
	cout << "调用函数模板" << endl;
}

template <typename T>
void func(T a, T b)
{
	cout << "调用函数模板重载" << endl;
}

int main()
{
	func(10);
	func<>(10);
	func(10, 20);
	func('a');
	//传入char类型，可隐式转换为int类型调用普通函数，但函数模板更匹配
	return 0;
}
```
