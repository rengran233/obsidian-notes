#include <iostream>

{
private:
	NT m_Name;
	AT m_Age;
public:
	Person(NT n, AT a)
	{
		m_Name = n;
		m_Age = a;
	}
	void show()
	{
		cout << m_Name << ' ' << m_Age << endl;

};

{
	Person<string, int> p("awa", 10);
	p.show();
	return 0;
}

类模板和函数模板的区别
1.类模板没有自动类型推导（C++17后可以？）
2.类模板的模板参数列表可以有默认参数（C++11后函数模板也支持默认参数）
```
#include <iostream>
using namespace std;

template <class NT, class AT = int>
class Person
{
private:
	NT m_Name;
	AT m_Age;
public:
	Person(NT n, AT a)
	{
		m_Name = n;
		m_Age = a;
	}
	void show()
	{
		cout << m_Name << ' ' << m_Age << endl;
	}
};

int main()
{
	Person<string, int> p1("awa", 10);
	Person<string> p2("qaq", 20);
	p1.show();
	p2.show();
	return 0;
}
```
