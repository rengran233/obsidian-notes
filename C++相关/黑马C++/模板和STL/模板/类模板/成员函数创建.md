类模板中的成员函数在调用时创建
```
#include <iostream>
using namespace std;

class Person1
{
public:
	void p1()
	{
		cout << 1 << endl;
	}
};

class Person2
{
public:
	void p2()
	{
		cout << 2 << endl;
	}
};

template <class T>
class tPerson
{
public:
	T obj;
	void func1()
	{
		obj.p1();
	}
	void func2()
	{
		obj.p2();
	}
};

int main()
{
	tPerson<Person2> pp;
	//pp.func1(); 可编译，不可运行，运行时调用func1，而odj为Person2类对象
	pp.func2();
	
	return 0;
}
```
