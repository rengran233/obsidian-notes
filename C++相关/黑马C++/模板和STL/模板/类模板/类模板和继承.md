#include <iostream>
using namespace std;

template <class T>
class base
{
	T a;
};

class son1 :public base<int> {}; //指定数据类型

template <class T1, class T2>
class son2 :public base<T1> //灵活指定父类T类型
{
	T2 b;
public:
	son2()
	{
		cout << typeid(T1).name() << endl;
		cout << typeid(T2).name() << endl;
	}
};

int main()
{
	son2<int, char> p;
	
	return 0;
}
