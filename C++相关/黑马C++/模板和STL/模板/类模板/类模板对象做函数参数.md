1.指定传入类型
2.参数模板化
3.类模板化
```
#include <iostream>
using namespace std;

template <class T1, class T2>
class Person
{
	T1 name;
	T2 age;
public:
	void show()
	{
		cout << this->name << ' ' << this->age << endl;
	}
	Person(T1 n, T2 a)
	{
		this->name = n;
		this->age = a;
	}
};

void print01(Person<string, int>& p) //传入指定类型
{
	p.show();
}

template <class T1, class T2>
void print02(Person<T1, T2>& p) //参数模板化
{
	p.show();
}

template <class T>
void print03(T& p) //类模板化
{
	p.show();
	cout << typeid(T).name();
}

int main()
{
	Person<string, int>p1("awa", 10);
	print01(p1);
	Person<string, int>p2("qaq", 20);
	print02(p2);
	Person<string, int>p3("owo", 30);
	print03(p3);
	
	return 0;
}
```
