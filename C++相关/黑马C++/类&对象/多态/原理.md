虚函数指针vfptr指向虚函数表vftable，子类重写覆盖父类虚函数表中的函数，父类vfptr指向重写后的函数


```cpp
#include <iostream>
using namespace std;

class AbsCal
{
public:
	int m_Num1;
	int m_Num2;
	virtual int getRes()
	{
		return 0;
	}
};

class AddCal :public AbsCal
{
public:
	int getRes()
	{
		return m_Num1 + m_Num2;
	}
};

class SubCal :public AbsCal
{
public:
	int getRes()
	{
		return m_Num1 - m_Num2;
	}
};

class MulCal :public AbsCal
{
public:
	int getRes()
	{
		return m_Num1 * m_Num2;
	}
};

void test01()
{
	AbsCal* abc = new AddCal;
	abc->m_Num1 = 10;
	abc->m_Num2 = 20;
	cout << abc->getRes() << endl;
	delete abc;
	abc = new MulCal;
	abc->m_Num1 = 2;
	abc->m_Num2 = 30;
	cout << abc->getRes() << endl;
	delete abc;
}

int main()
{
	test01();
	return 0;
}
```
