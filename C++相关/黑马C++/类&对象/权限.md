1.控制读写权限
```
class Person{
public:
	string getName(){
		return m_Name;
	}
	void setName(string name){
		m_Name = name;
	}
	int getAge(){
		return m_Age;
	}
	void setIncome(int income){
		m_Income = income;
	}
private:
	string m_Name; //可读可写
	int m_Age; //可读不可写
	int m_Income; //不可读可写
};
```

2.过滤有效数据
```
class Person{
public:
	void setAge(int age){
		if(age < 0 || age > 150) //判断年龄是否属于0~150
			return;
		m_Age = age;
	}
};
```
