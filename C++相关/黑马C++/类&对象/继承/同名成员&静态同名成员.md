子类和父类**可以有同名成员**，在子类中该名称会调用子类中成员，需要在父类**作用域**下调用父类成员
```
//调用同名成员属性
#include <iostream>
using namespace std;

class father
{
public:
	int m_A = 10;
};

class son :public father
{
public:
	int m_A = 20;
};

void test01()
{
	son s;
	cout << s.m_A << endl; //子类成员20
	cout << s.father::m_A << endl; //父类成员10
}

int main()
{
	test01();
	return 0;
}
```


如果子类中出现与父类同名的成员函数，会隐藏所有父类中的同名成员函数
```
//调用同名成员函数
#include <iostream>
using namespace std;

class father
{
public:
	void func()
	{
		cout << "awa" << endl;
	}
	void func(int)
	{
		cout << "qwq" << endl;
	}
};

class son :public father
{
public:
	void func()
	{
		cout << "qaq" << endl;
	}
};

void test01()
{
	son s;
	s.func();
	//s.func(10); 无法运行，子类中无函数func(int)
	s.father::func();
	s.father::func(10);
}

int main()
{
	test01();
	return 0;
}
```


静态同名成员与非静态调用方法类似，不过有两种访问方式
```
#include <iostream>
using namespace std;

class base
{
public:
	static int m_A;
};
int base::m_A = 10;

class son :public base
{
public:
	static int m_A;
};
int son::m_A = 20;

int main()
{
	cout << son().m_A << endl;
	cout << son().base::m_A << endl;
	cout << son::m_A << endl;
	cout << son::base::m_A << endl;
	return 0;
}
```
