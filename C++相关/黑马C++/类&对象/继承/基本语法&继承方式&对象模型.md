继承的好处：减少重复代码
`class <子类> : [继承方式] <父类>`
子类也称为派生类，父类也称为基类。
```
#include <iostream>
#include <string>
using namespace std;

class BasePage
{
public:
	void header();
	void footer();
	void left();
};

class Java : public BasePage
{
public:
	void content();
};

class Python : public BasePage
{
public:
	void content();
};

class CPP : public BasePage
{
public:
	void content();
};

void BasePage::header()
{
	cout << "公共头部" << endl;
}

void BasePage::footer()
{
	cout << "公共底部" << endl;
}

void BasePage::left()
{
	cout << "全部分类" << endl;
}

void Java::content()
{
	cout << "Java视频内容" << endl;
}

void Python::content()
{
	cout << "Python视频内容" << endl;
}

void CPP::content()
{
	cout << "CPP视频内容" << endl;
}

void ShowJava()
{
	Java ja;
	ja.header();
	ja.footer();
	ja.left();
	ja.content();
	cout << "--------------" << endl;
}

void ShowPython()
{
	Python py;
	py.header();
	py.footer();
	py.left();
	py.content();
	cout << "--------------" << endl;
}

void ShowCPP()
{
	CPP cp;
	cp.header();
	cp.footer();
	cp.left();
	cp.content();
	cout << "--------------" << endl;
}

int main()
{
	ShowJava();
	ShowPython();
	ShowCPP();
	return 0;
}
```


三种继承方式：**公共继承、保护继承、私有继承**
```
class A //父类
{
public:
	int m_A;
protected:
	int m_B;
private:
	int m_C;
};

//公共继承
class B1 : public A
{
public:
	int m_A;
protected:
	int m_B;
/*不可访问:
	int m_C;*/
};

//保护继承
class B2 : protected A
{
protected:
	int m_A;
	int m_B;
/*不可访问：
	int m_C; */
};

//私有继承
class B3 : private A
{
private:
	int m_A;
	int m_B;
/*不可访问:
	int m_C; */
}
```


子类会继承父类的所有属性，但不能访问父类的私有属性
```
class A
{
public:
	int m_A;
private:
	int m_B;
protected:
	int m_C;
};

class B :public A
{
	int m_D;
};

int main()
{
	cout << sizeof(B); //输出16，4个int数据
}
```
