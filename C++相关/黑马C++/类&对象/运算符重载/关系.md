重载`==`运算符
```
#include <iostream>
#include <string>
using namespace std;

class Person
{
public:
	string m_Name;
	int m_Age;
	Person(string n, int a)
	{
		m_Name = n;
		m_Age = a;
	}
	bool operator==(Person& p)
	{
		return (m_Name == p.m_Name && m_Age == p.m_Age);
	}
};

int main()
{
	Person p1("awa", 10);
	Person p2("awa", 10);
	if (p1 == p2)
		cout << "==" << endl;
	else
		cout << "!=" << endl;
	return 0;
}
```
