#include <iostream>
using namespace std;

class MyInteger
{
	friend ostream &operator<<(ostream& cout, MyInteger const &mi);
public:
	MyInteger(int a)
	{
		m_Num = a;
	}
	MyInteger &operator++()
	{
		m_Num++;
		return *this; //将自身返回，可以重复递增
	}
	MyInteger operator++(int) //占位
	{
		MyInteger temp = *this;
		m_Num++;
		return temp;
	}
private:
	int m_Num;
};

ostream& operator<<(ostream& coout, MyInteger const &mi)
//const?
{
	cout << mi.m_Num;
	return cout;
}

int main()
{
	MyInteger myint(0);
	cout << ++(++myint) << endl;
	cout << myint << endl;
	cout << myint++ << endl; //返回右值，不可链式编程
	cout << myint << endl;
	return 0;
}
