`#include<cctype>`
- `isalpha(c)`：检测字母（a-z, A-Z）
- `isdigit(c)`：检测数字（0-9）
- `isspace(c)`：检测空白字符（空格、\t、\n等）
- `isalnum(c)`：检测字母或数字
- `tolower(c)`/`toupper(c)`：大小写转换


```cpp
#include <sstream>
#include <iostream>

int main() {
	//istringstream类
    std::string input = "42 3.14 hello";
    std::istringstream iss(input);  // 将字符串绑定到输入流
    int num;
    double pi;
    std::string text;
    iss >> num >> pi >> text;  // 依次提取整数、浮点数、字符串
    std::cout << num << ", " << pi << ", " << text;  // 输出：42, 3.14, hello
	iss.str("new content");  // 通过str()重置输入流内容
    //支持所有内置类型自动转换，默认以空格为分隔符

	//ostringstream类
	std::ostringstream oss;
	oss << num << ' ' << pi;
	std::string result = oss.str();
	oss.str(""); //清空流内容
    return 0;
}
```
