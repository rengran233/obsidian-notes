## 二分查找
```cpp
// 升序数组二分查找
int BinarySearch(vector<int>& arr, int target) {
	int left = 0, right = arr.size() - 1;
	while (left <= right) {
		//可以防止left+right过大溢出
		int mid = left + (right - left) / 2;
		if (arr[mid] < target) left = mid + 1;
		else if (arr[mid] > target) right = mid - 1;
		else return mid;
	}
	return -1;
}
```
