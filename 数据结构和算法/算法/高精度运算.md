string hmultiple(string s1, string s2)
{
	reverse(s1.begin(), s1.end());
	reverse(s2.begin(), s2.end());
	int s1s = s1.size(), s2s = s2.size();
	vector<int> res(s1s + s2s);
	for (int i = 0; i < s1s; ++i)
		for (int j = 0; j < s2s; ++j)
			res[i + j] += (s1[i] - '0') * (s2[j] - '0');
	for (int i = 0; i < s1s + s2s; ++i)
	{
		if (res[i] >= 10)
		{
			res[i + 1] += res[i] / 10;
			res[i] %= 10;
		}
	}
	int len = s1s + s2s;
	while (res[len - 1] == 0 && len > 1) len--;
	string sres;
	for (int i = len - 1; i >= 0; --i) 
		sres += char(res[i] + '0');
	return sres;
}

string hplus(string s1, string s2)
{
	reverse(s1.begin(), s1.end());
	reverse(s2.begin(), s2.end());
	int s1s = s1.size(), s2s = s2.size();
	s1s >= s2s ? s2.resize(s1s, '0') : s1.resize(s2s, '0');
	int reslen = s1.size() + 1;
	vector<int> res(reslen);
	for (int i = 0; i < reslen - 1; ++i)
		res[i] = (s1[i] - '0') + (s2[i] - '0');
	for (int i = 0; i < reslen - 1; ++i)
	{
		if (res[i] >= 10)
		{
			res[i + 1] += res[i] / 10;
			res[i] %= 10;
		}
	}
	while (res[reslen - 1] == 0 && reslen > 1) reslen--;
	string sres;
	for (int i = reslen - 1; i >= 0; --i)
		sres += char(res[i] + '0');
	return sres;
}
