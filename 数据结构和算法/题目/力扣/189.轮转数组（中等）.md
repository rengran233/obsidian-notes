### 作答（建新数组）：
```cpp
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        if (k == 0) return;
        int n = nums.size();
        //新数组首元素位置
        //k%n防止k过大使n-k为负数，对负数取模结果为负数
        //对整体取模防止k=n时pos=n使数组越界
        int pos = (n - k % n) % n;
        vector<int> res;
        for (int i = 0; i < n; i++) {
            res.push_back(nums[pos]);
            pos = (pos + 1) % n;
        }
        nums = res;
    }
};
```
时间复杂度：$O(n)$
空间复杂度：$O(n)$


### 数组翻转：
```cpp
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        if (k == 0) return;
        //数组整体翻转
        reverse(nums.begin(), nums.end());
        auto it = nums.begin();
        advance(it, k % nums.size());
        //轮转部分反转
        reverse(nums.begin(), it);
        //剩余部分翻转
        reverse(it, nums.end());
    }
};
```
$[1,2,3,4,5]$ -> $[5,4,3,2,1]$ ->$[3,4,5,1,2]$，轮转三次
时间复杂度：$O(n)$
空间复杂度：$O(1)$
