快慢指针法：
```cpp
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        if (head == nullptr || head->next == nullptr) return head;
        ListNode* slow = head, * fast = head->next;
        while (fast->next != nullptr && fast->next->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }
        return slow->next;
    }
};
```
若节点个数为奇数，取到中间节点前一个；若为偶数，取到前半段最后一节点。


递归法：
```cpp
class Solution {
public:
    int cnt = 0, rcnt = 0;
    ListNode* md = new ListNode;
    ListNode* middleNode(ListNode* head) {
        if (head == nullptr) return head;
        cnt++;
        middleNode(head->next);
        rcnt++;
        if (rcnt == (cnt + 1) / 2) md = head;
        return md;
    }
};
```
