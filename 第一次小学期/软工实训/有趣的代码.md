```cpp
#pragma once
#include <QObject> // 包含QObject以使用Q_ENUM
class GameState {
    Q_GADGET // 使用Q_GADGET而不是Q_OBJECT，因为它没有信号槽，更轻量
    Q_ENUMS(State)
public:
    enum State {
        MainMenu,      // 主菜单状态
        LoadingLevel,  // 关卡加载中
        Playing,       // 游戏中
        Paused,        // 暂停
        PlayerWon,     // 玩家胜利
        GameOver       // 游戏结束
    };
private:
    // 防止被实例化
    GameState();
};
```
用一个单独的类包装出一个枚举
